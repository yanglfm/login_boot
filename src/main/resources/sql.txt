create table module
(
    id          int auto_increment
        primary key,
    name        varchar(100) null comment '模块名称
',
    path        varchar(100) null comment '模块路径',
    icon        varchar(100) null comment '图标',
    module_type varchar(50)  null comment '模块类型：页面模块还是侧边栏模块：
0：侧边栏模块
1：页面模块',
    order_no    int          null comment '模块的顺序',
    parent_id   int          null comment '父类id，如果是侧边栏的主要模块，为父模块，值为空，表示侧边栏的几个大模块
',
    component   varchar(100) null comment '字符串格式的组件地址',
    redirect    varchar(100) null comment '转发路径'
)
    comment '模块表';

create table role
(
    id   int auto_increment
        primary key,
    name varchar(50) null comment '角色名
'
)
    comment '角色表';

create table role_module
(
    id        int auto_increment
        primary key,
    role_id   int null,
    module_id int null
)
    comment '角色和模块关系表';

create table user
(
    user_id      int auto_increment
        primary key,
    name         varchar(36)                         null comment '名称',
    avatar       varchar(255)                        null comment '微信头像',
    phone        varchar(11)                         null comment '手机号',
    gender       int                                 null comment '性别',
    open_id      varchar(50)                         null comment '微信openid',
    union_id     varchar(50)                         null comment '微信union_id',
    id_card_no   varchar(50)                         null comment '身份证号码',
    id_card_name varchar(50)                         null comment '身份证姓名',
    create_time  timestamp default CURRENT_TIMESTAMP null comment '创建时间',
    update_time  timestamp default CURRENT_TIMESTAMP null comment '修改时间',
    country      varchar(50)                         null comment '国家',
    province     varchar(50)                         null comment '省份',
    city         varchar(50)                         null comment '城市',
    language     varchar(50)                         null comment '语言',
    status       int       default 1                 null comment '用户状态：1：存在 0:删除',
    password     varchar(50)                         null
);

create table user_role
(
    id      int auto_increment
        primary key,
    user_id varchar(50) null comment '用户id',
    role_id varchar(50) null comment '角色id'
)
    comment '用户和角色关系表';

create table web_log
(
    id           int auto_increment comment 'log_id'

',
    user_id      int          null comment '用户id',
    start_time   timestamp    null comment '用户访问的起始时间'
)
    comment '用户操作日志表';
